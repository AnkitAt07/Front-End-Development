Switch Statement.

Syntax:

switch (expression) 
    {
        case value1:
            // Code to execute when expression === value1
            break;
        case value2:
            // Code to execute when expression === value2
            break;
        case value3:
            // Code to execute when expression === value3
            break;
        case value4:
            // Code to execute when expression === value3
            break;
        default:
            // Code to execute if no cases match
    }
